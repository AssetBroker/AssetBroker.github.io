    #include "ListenerBridge.hpp"

#include <algorithm>
#include <chrono>
#include <exception>
#include <iostream>
#include <stdexcept>
#include <thread>

namespace ListenerBridgeApp
{
    ListenerBridge::ListenerBridge(ListenerBridgeConfig config,
        std::unique_ptr<PluginSystem::AProtocol> protocol)
        : _config(std::move(config))
        , _sessionManager(_config.listenerName)
        , _protocol(std::move(protocol))
        , _unixReadBuffer()
        , _protocolReadBuffer()
        , _pendingResponses(0)
        , _protocolReady(false)
        , _stopRequested(false)
        , _reconnectDelay(_config.reconnectInitialMs)
        , _nextReconnectAttempt(std::chrono::steady_clock::now())
    {
        if (_config.sessionId.empty())
            throw std::invalid_argument("ListenerBridge requires a non-empty session id");
        if (_config.unixReadMaxBytes == 0)
            throw std::invalid_argument("ListenerBridge unixReadMaxBytes must be > 0");
        if (_config.protocolReadMaxBytes == 0)
            throw std::invalid_argument("ListenerBridge protocolReadMaxBytes must be > 0");
        if (_config.idleSleepMs <= 0)
            throw std::invalid_argument("ListenerBridge idleSleepMs must be > 0");
        if (_config.reconnectInitialMs <= 0 || _config.reconnectMaxMs < _config.reconnectInitialMs)
            throw std::invalid_argument("ListenerBridge reconnect settings are invalid");
        if (!_protocol)
            throw std::invalid_argument("ListenerBridge requires a protocol instance");

        _sessionManager.setActiveSession(_config.sessionId);

        std::cout << "[ListenerBridge] IPC socket: "
                  << _sessionManager.getSessionPath(_config.sessionId) << '\n';
    }

    void ListenerBridge::run()
    {
        while (!_stopRequested) {
            bool didWork = false;

            if (!ensureProtocolConnected()) {
                std::this_thread::sleep_for(std::chrono::milliseconds(_config.idleSleepMs));
                continue;
            }

            try {
                didWork = pumpUnixToProtocol() || didWork;
            } catch (const std::exception &error) {
                handleProtocolError("forwarding unix->protocol", error);
            }

            if (_pendingResponses > 0) {
                try {
                    didWork = pumpProtocolToUnix() || didWork;
                } catch (const std::exception &error) {
                    handleProtocolError("forwarding protocol->unix", error);
                }
            }

            if (!didWork)
                std::this_thread::sleep_for(std::chrono::milliseconds(_config.idleSleepMs));
        }

        try {
            _protocol->disconnect();
            _protocolReady = false;
        } catch (const std::exception &error) {
            std::cerr << "[ListenerBridge] Protocol disconnect failed: " << error.what() << '\n';
        }
    }

    void ListenerBridge::requestStop()
    {
        _stopRequested = true;
    }

    bool ListenerBridge::ensureProtocolConnected()
    {
        if (_protocolReady)
            return true;

        const auto now = std::chrono::steady_clock::now();
        if (now < _nextReconnectAttempt)
            return false;

        try {
            _protocol->connect();
            _protocolReady = true;
            _reconnectDelay = std::chrono::milliseconds(_config.reconnectInitialMs);
            _nextReconnectAttempt = now;
            std::cout << "[ListenerBridge] Protocol connected" << '\n';
            return true;
        } catch (const std::exception &error) {
            std::cerr << "[ListenerBridge] Protocol connect failed: " << error.what()
                      << " (retry in " << _reconnectDelay.count() << "ms)\n";
            _nextReconnectAttempt = now + _reconnectDelay;
            _reconnectDelay = std::min(
                std::chrono::milliseconds(_config.reconnectMaxMs),
                _reconnectDelay * 2);
            return false;
        }
    }

    void ListenerBridge::handleProtocolError(const std::string &context, const std::exception &error)
    {
        std::cerr << "[ListenerBridge] Error while " << context << ": " << error.what() << '\n';

        try {
            _protocol->disconnect();
            _protocolReady = false;
        } catch (const std::exception &disconnectError) {
            std::cerr << "[ListenerBridge] Protocol disconnect failed: "
                      << disconnectError.what() << '\n';
        }

        const auto now = std::chrono::steady_clock::now();
        _nextReconnectAttempt = now + _reconnectDelay;
        _reconnectDelay = std::min(
            std::chrono::milliseconds(_config.reconnectMaxMs),
            _reconnectDelay * 2);
        _pendingResponses = 0;
        _protocolReadBuffer.clear();
    }

    bool ListenerBridge::pumpUnixToProtocol()
    {
        std::string chunk = _sessionManager.readActive(_config.unixReadMaxBytes, true);
        if (!chunk.empty())
            _unixReadBuffer += chunk;

        bool didWork = false;
        std::string line;
        while (popLine(_unixReadBuffer, line)) {
            _protocol->send(line + "\n");
            ++_pendingResponses;
            didWork = true;
        }
        return didWork;
    }

    bool ListenerBridge::pumpProtocolToUnix()
    {
        std::string chunk = _protocol->receive(_config.protocolReadMaxBytes, true);
        if (!chunk.empty())
            _protocolReadBuffer += chunk;

        bool didWork = false;
        std::string line;
        while (popLine(_protocolReadBuffer, line)) {
            _sessionManager.writeActive(line + "\n");
            if (_pendingResponses > 0)
                --_pendingResponses;
            didWork = true;
        }
        return didWork;
    }

    bool ListenerBridge::popLine(std::string &buffer, std::string &line)
    {
        std::size_t newlinePos = buffer.find('\n');
        if (newlinePos == std::string::npos)
            return false;

        line = buffer.substr(0, newlinePos);
        buffer.erase(0, newlinePos + 1);
        return true;
    }
