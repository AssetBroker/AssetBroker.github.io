    #include "ListenerBridge.hpp"

#include <atomic>
#include <csignal>
#include <cstdint>
#include <dlfcn.h>
#include <exception>
#include <filesystem>
#include <iostream>
#include <memory>
#include <stdexcept>
#include <string>
#include <unordered_set>
#include <vector>
#include <algorithm>

#include "PluginSystem/HotReloadSystem.hpp"
#include "json/json.hpp"

namespace {
    struct CliOptions
    {
        std::string listenerName = "default";
        std::string sessionId;
        std::string sessionsRootDir = "/sessions";
        std::string protocolsDir = "/plugin/protocols";
        std::string proto;
        std::string host;
        std::string bindHost;
        std::uint16_t port = 0;
    };

    struct ProtocolRuntime
    {
        std::unique_ptr<PluginSystem::AProtocol> protocol;
        void *pluginHandle = nullptr;
    };

    using CreateProtocolFn = PluginSystem::AProtocol* (*)(const char *configJson);

    static ListenerBridgeApp::ListenerBridge *g_bridge = nullptr;
    static std::atomic<bool> g_stopRequested{false};
    static std::atomic<bool> g_reloadRequested{false};

    class ProtocolHotReloadWatcher: public HotReloadSystem
    {
        public:
            ProtocolHotReloadWatcher(const std::string &directory,
                const std::vector<std::string> &watchedNames)
                : HotReloadSystem(directory)
                , _watchedNames(watchedNames.begin(), watchedNames.end())
            {
            }

        protected:
            void onFileCreated(const std::string &filename) override
            {
                triggerReload(filename, "created");
            }

            void onFileModified(const std::string &filename) override
            {
                triggerReload(filename, "modified");
            }

            void onFileDeleted(const std::string &filename) override
            {
                triggerReload(filename, "deleted");
            }

        private:
            void triggerReload(const std::string &filename, const char *event)
            {
                if (_watchedNames.find(filename) == _watchedNames.end())
                    return;

                std::cout << "[ListenerBridge] Protocol plugin file " << event
                          << ": " << filename << ", scheduling reload\n";

                g_reloadRequested = true;
                if (g_bridge)
                    g_bridge->requestStop();
            }

            std::unordered_set<std::string> _watchedNames;
    };

    void printUsage(const char *binary)
    {
        std::cout
            << "Usage:\n"
            << "  " << binary << " --session <id> --proto <protocol-plugin-name> [options]\n\n"
            << "Options:\n"
            << "  --listener <name>     Listener namespace for <sessions-root>/<name>/... (default: default)\n"
            << "  --session <id>        Session id used for session_<id>.sock (required)\n"
            << "  --sessions-root <dir> Root directory for unix socket sessions (default: /sessions)\n"
            << "  --protocols-dir <dir> Directory containing protocol .so plugins (default: /plugin/protocols)\n"
            << "  --proto <name>        Protocol plugin name, e.g. tcp-bind or tcp_bind (required)\n"
            << "  --host <ip>           Host value passed to protocol config\n"
            << "  --bind <ip>           Bind value passed to protocol config\n"
            << "  --port <port>         Port for selected protocol\n"
            << "  --help                Show this help\n";
    }

    bool parsePort(const std::string &value, std::uint16_t &outPort)
    {
        try {
            std::size_t idx = 0;
            unsigned long parsed = std::stoul(value, &idx);
            if (idx != value.size() || parsed == 0 || parsed > 65535)
                return false;
            outPort = static_cast<std::uint16_t>(parsed);
            return true;
        } catch (...) {
            return false;
        }
    }

    bool readArgValue(int argc, char **argv, int &index, std::string &out)
    {
        if (index + 1 >= argc)
            return false;
        ++index;
        out = argv[index];
        return true;
    }

    bool parseArgs(int argc, char **argv, CliOptions &options)
    {
        for (int i = 1; i < argc; ++i) {
            std::string arg = argv[i];
            if (arg == "--help") {
                printUsage(argv[0]);
                return false;
            }
            if (arg == "--listener") {
                if (!readArgValue(argc, argv, i, options.listenerName))
                    throw std::runtime_error("Missing value for --listener");
                continue;
            }
            if (arg == "--session") {
                if (!readArgValue(argc, argv, i, options.sessionId))
                    throw std::runtime_error("Missing value for --session");
                continue;
            }
            if (arg == "--sessions-root") {
                if (!readArgValue(argc, argv, i, options.sessionsRootDir))
                    throw std::runtime_error("Missing value for --sessions-root");
                continue;
            }
            if (arg == "--protocols-dir") {
                if (!readArgValue(argc, argv, i, options.protocolsDir))
                    throw std::runtime_error("Missing value for --protocols-dir");
                continue;
            }
            if (arg == "--proto") {
                if (!readArgValue(argc, argv, i, options.proto))
                    throw std::runtime_error("Missing value for --proto");
                continue;
            }
            if (arg == "--host") {
                if (!readArgValue(argc, argv, i, options.host))
                    throw std::runtime_error("Missing value for --host");
                continue;
            }
            if (arg == "--bind") {
                if (!readArgValue(argc, argv, i, options.bindHost))
                    throw std::runtime_error("Missing value for --bind");
                continue;
            }
            if (arg == "--port") {
                std::string portString;
                if (!readArgValue(argc, argv, i, portString))
                    throw std::runtime_error("Missing value for --port");
                if (!parsePort(portString, options.port))
                    throw std::runtime_error("Invalid --port value: " + portString);
                continue;
            }
            throw std::runtime_error("Unknown argument: " + arg);
        }

        if (options.sessionId.empty())
            throw std::runtime_error("--session is required");
        if (options.proto.empty())
            throw std::runtime_error("--proto is required");
        if (options.port == 0)
            throw std::runtime_error("--port is required and must be > 0");

        if (options.host.empty())
            options.host = "0.0.0.0";
        if (options.bindHost.empty())
            options.bindHost = "0.0.0.0";

        return true;
    }

    std::string toDashed(std::string value)
    {
        for (auto &ch : value) {
            if (ch == '_')
                ch = '-';
        }
        return value;
    }

    std::string toUnderscored(std::string value)
    {
        for (auto &ch : value) {
            if (ch == '-')
                ch = '_';
        }
        return value;
    }

    std::vector<std::string> protocolLibraryCandidates(const std::string &name)
    {
        std::vector<std::string> candidates;

        const std::string dashed = toDashed(name);
        const std::string underscored = toUnderscored(name);
        const std::string protocolSuffix = dashed + "-protocol";

        auto pushUnique = [&candidates](const std::string &value) {
            if (value.empty())
                return;
            if (std::find(candidates.begin(), candidates.end(), value) == candidates.end())
                candidates.push_back(value);
        };

        pushUnique(name);
        pushUnique(dashed);
        pushUnique(underscored);
        pushUnique(protocolSuffix);

        std::vector<std::string> asFiles;
        asFiles.reserve(candidates.size() * 2);
        for (const auto &candidate : candidates) {
            asFiles.push_back(candidate);
            asFiles.push_back(candidate + ".so");
        }

        return asFiles;
    }

    nlohmann::json protocolConfigJson(const CliOptions &options)
    {
        const std::string protoKey = toUnderscored(options.proto);
        const std::string host = protoKey == "tcp_bind" ? options.bindHost : options.host;
        return nlohmann::json{
            {"host", host},
            {"bind", options.bindHost},
            {"port", options.port}
        };
    }

    ProtocolRuntime createProtocol(const CliOptions &options)
    {
        namespace fs = std::filesystem;

        const fs::path protocolsPath(options.protocolsDir);
        const auto candidates = protocolLibraryCandidates(options.proto);
        const std::string configDump = protocolConfigJson(options).dump();

        for (const auto &candidate : candidates) {
            const fs::path libraryPath = protocolsPath / candidate;
            if (!fs::exists(libraryPath) || !fs::is_regular_file(libraryPath))
                continue;

            void *handle = ::dlopen(libraryPath.string().c_str(), RTLD_NOW);
            if (!handle) {
                std::cerr << "[ListenerBridge] dlopen failed for " << libraryPath
                          << ": " << ::dlerror() << '\n';
                continue;
            }

            ::dlerror();
            auto createFn = reinterpret_cast<CreateProtocolFn>(::dlsym(handle, "create_protocol"));
            const char *createError = ::dlerror();
            if (createError) {
                std::cerr << "[ListenerBridge] Missing create_protocol in " << libraryPath
                          << ": " << createError << '\n';
                ::dlclose(handle);
                continue;
            }

            PluginSystem::AProtocol *raw = nullptr;
            try {
                raw = createFn(configDump.c_str());
            } catch (const std::exception &error) {
                std::cerr << "[ListenerBridge] create_protocol threw for " << libraryPath
                          << ": " << error.what() << '\n';
                ::dlclose(handle);
                continue;
            }

            if (!raw) {
                std::cerr << "[ListenerBridge] create_protocol returned null for "
                          << libraryPath << '\n';
                ::dlclose(handle);
                continue;
            }

            std::unique_ptr<PluginSystem::AProtocol> protocol(raw);
            protocol->toggle();

            std::cout << "[ListenerBridge] Loaded protocol plugin: " << libraryPath << '\n';

            ProtocolRuntime runtime;
            runtime.protocol = std::move(protocol);
            runtime.pluginHandle = handle;
            return runtime;
        }

        throw std::runtime_error(
            "Unable to load protocol plugin for '" + options.proto +
            "' from directory: " + options.protocolsDir);
    }

    extern "C" void handleSignal(int)
    {
        g_stopRequested = true;
        if (g_bridge)
            g_bridge->requestStop();
    }
}

int main(int argc, char **argv)
{
    CliOptions options;

    try {
        if (!parseArgs(argc, argv, options))
            return 0;

        ListenerBridgeApp::ListenerBridgeConfig bridgeConfig;
        bridgeConfig.listenerName = options.listenerName;
        bridgeConfig.sessionId = options.sessionId;
        bridgeConfig.sessionsRootDir = options.sessionsRootDir;

        std::signal(SIGINT, handleSignal);
        std::signal(SIGTERM, handleSignal);

        const auto watchCandidates = protocolLibraryCandidates(options.proto);
        ProtocolHotReloadWatcher hotReload(options.protocolsDir, watchCandidates);
        hotReload.start();

        while (!g_stopRequested) {
            auto runtime = createProtocol(options);
            ListenerBridgeApp::ListenerBridge bridge(bridgeConfig, std::move(runtime.protocol));
            g_bridge = &bridge;

            std::cout << "[ListenerBridge] Started with protocol=" << options.proto
                      << ", listener=" << options.listenerName
                      << ", session=" << options.sessionId
                      << ", port=" << options.port
                      << '\n';

            bridge.run();
            g_bridge = nullptr;

            if (runtime.pluginHandle)
                ::dlclose(runtime.pluginHandle);

            if (g_reloadRequested.exchange(false)) {
                std::cout << "[ListenerBridge] Reloading protocol plugin...\n";
                continue;
            }

            if (!g_stopRequested)
                break;
        }

        hotReload.stop();

        return 0;
    } catch (const std::exception &error) {
        std::cerr << "listener-bridge error: " << error.what() << '\n';
        printUsage(argv[0]);
        return 1;
    }
}
