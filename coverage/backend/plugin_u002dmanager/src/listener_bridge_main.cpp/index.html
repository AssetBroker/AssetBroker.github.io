    #include "ListenerBridge.hpp"

#include <csignal>
#include <cstdint>
#include <exception>
#include <iostream>
#include <memory>
#include <stdexcept>
#include <string>

#include "TCPBindProtocol.hpp"
#include "TCPConnectProtocol.hpp"
#include "UDPProtocol.hpp"
#include "json/json.hpp"

namespace {
    struct CliOptions
    {
        std::string listenerName = "default";
        std::string sessionId;
        std::string proto;
        std::string host;
        std::string bindHost;
        std::uint16_t port = 0;
    };

    static ListenerBridgeApp::ListenerBridge *g_bridge = nullptr;

    void printUsage(const char *binary)
    {
        std::cout
            << "Usage:\n"
            << "  " << binary << " --session <id> --proto tcp_connect|tcp_bind|udp [options]\n\n"
            << "Options:\n"
            << "  --listener <name>     Listener namespace for /tmp/listeners/<name>/... (default: default)\n"
            << "  --session <id>        Session id used for session_<id>.sock (required)\n"
            << "  --proto <name>        tcp_connect | tcp_bind | udp (required)\n"
            << "  --host <ip>           Remote host for tcp_connect/udp\n"
            << "  --bind <ip>           Bind host for tcp_bind\n"
            << "  --port <port>         Port for selected protocol\n"
            << "  --help                Show this help\n";
    }

    bool parsePort(const std::string &value, std::uint16_t &outPort)
    {
        try {
            std::size_t idx = 0;
            unsigned long parsed = std::stoul(value, &idx);
            if (idx != value.size() || parsed == 0 || parsed > 65535)
                return false;
            outPort = static_cast<std::uint16_t>(parsed);
            return true;
        } catch (...) {
            return false;
        }
    }

    bool readArgValue(int argc, char **argv, int &index, std::string &out)
    {
        if (index + 1 >= argc)
            return false;
        ++index;
        out = argv[index];
        return true;
    }

    bool parseArgs(int argc, char **argv, CliOptions &options)
    {
        for (int i = 1; i < argc; ++i) {
            std::string arg = argv[i];
            if (arg == "--help") {
                printUsage(argv[0]);
                return false;
            }
            if (arg == "--listener") {
                if (!readArgValue(argc, argv, i, options.listenerName))
                    throw std::runtime_error("Missing value for --listener");
                continue;
            }
            if (arg == "--session") {
                if (!readArgValue(argc, argv, i, options.sessionId))
                    throw std::runtime_error("Missing value for --session");
                continue;
            }
            if (arg == "--proto") {
                if (!readArgValue(argc, argv, i, options.proto))
                    throw std::runtime_error("Missing value for --proto");
                continue;
            }
            if (arg == "--host") {
                if (!readArgValue(argc, argv, i, options.host))
                    throw std::runtime_error("Missing value for --host");
                continue;
            }
            if (arg == "--bind") {
                if (!readArgValue(argc, argv, i, options.bindHost))
                    throw std::runtime_error("Missing value for --bind");
                continue;
            }
            if (arg == "--port") {
                std::string portString;
                if (!readArgValue(argc, argv, i, portString))
                    throw std::runtime_error("Missing value for --port");
                if (!parsePort(portString, options.port))
                    throw std::runtime_error("Invalid --port value: " + portString);
                continue;
            }
            throw std::runtime_error("Unknown argument: " + arg);
        }

        if (options.sessionId.empty())
            throw std::runtime_error("--session is required");
        if (options.proto.empty())
            throw std::runtime_error("--proto is required");
        if (options.port == 0)
            throw std::runtime_error("--port is required and must be > 0");

        if (options.proto == "tcp_connect" || options.proto == "udp") {
            if (options.host.empty())
                throw std::runtime_error("--host is required for " + options.proto);
        } else if (options.proto == "tcp_bind") {
            if (options.bindHost.empty())
                throw std::runtime_error("--bind is required for tcp_bind");
        } else {
            throw std::runtime_error("Unsupported --proto value: " + options.proto);
        }

        return true;
    }

    std::unique_ptr<PluginSystem::AProtocol> createProtocol(const CliOptions &options)
    {
        if (options.proto == "tcp_connect") {
            auto protocol = std::make_unique<PluginSystem::TCPConnectProtocol>(
                nlohmann::json{{"host", options.host}, {"port", options.port}});
            protocol->toggle();
            return protocol;
        }

        if (options.proto == "tcp_bind") {
            auto protocol = std::make_unique<PluginSystem::TCPBindProtocol>(
                nlohmann::json{{"host", options.bindHost}, {"port", options.port}});
            protocol->toggle();
            return protocol;
        }

        auto protocol = std::make_unique<PluginSystem::UDPProtocol>(
            nlohmann::json{{"host", options.host}, {"port", options.port}});
        protocol->toggle();
        return protocol;
    }

    extern "C" void handleSignal(int)
    {
        if (g_bridge)
            g_bridge->requestStop();
    }
}

int main(int argc, char **argv)
{
    CliOptions options;

    try {
        if (!parseArgs(argc, argv, options))
            return 0;

        ListenerBridgeApp::ListenerBridgeConfig bridgeConfig;
        bridgeConfig.listenerName = options.listenerName;
        bridgeConfig.sessionId = options.sessionId;

        auto protocol = createProtocol(options);
        ListenerBridgeApp::ListenerBridge bridge(bridgeConfig, std::move(protocol));
        g_bridge = &bridge;

        std::signal(SIGINT, handleSignal);
        std::signal(SIGTERM, handleSignal);

        std::cout << "[ListenerBridge] Started with protocol=" << options.proto
                  << ", listener=" << options.listenerName
                  << ", session=" << options.sessionId
                  << ", port=" << options.port
                  << '\n';

        bridge.run();
        g_bridge = nullptr;
        return 0;
    } catch (const std::exception &error) {
        std::cerr << "listener-bridge error: " << error.what() << '\n';
        printUsage(argv[0]);
        return 1;
    }
}
