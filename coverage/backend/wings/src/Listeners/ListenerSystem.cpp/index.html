    std::string normalizeProtocolName(std::string plugin_name)
    {
      const auto slash_pos = plugin_name.find_last_of("/\\");
      if (slash_pos != std::string::npos)
        plugin_name = plugin_name.substr(slash_pos + 1);

      if (plugin_name.size() > 3 &&
        plugin_name.substr(plugin_name.size() - 3) == ".so") {
        plugin_name = plugin_name.substr(0, plugin_name.size() - 3);
      }

      std::replace(plugin_name.begin(), plugin_name.end(), '_', '-');

      return plugin_name;
    }

    std::string underscoredProtocolName(std::string value)
    {
      std::replace(value.begin(), value.end(), '-', '_');
      return value;
    }

    std::string dockerPortToNumeric(const std::string& docker_port)
    {
      const auto slash_pos = docker_port.find('/');
      if (slash_pos == std::string::npos)
        return docker_port;
      return docker_port.substr(0, slash_pos);
    }

    std::string sanitizeSessionId(std::string value)
    {
      std::replace(value.begin(), value.end(), '/', '-');
      std::replace(value.begin(), value.end(), '\\', '-');
      std::replace(value.begin(), value.end(), ' ', '-');
      return value;
    }

    std::string resolveSessionsSourcePath()
    {
      namespace fs = std::filesystem;

      const char* configured = std::getenv("ASSET_BROKER_SESSIONS_DIR");
      fs::path sessions_path;

      auto normalizeForDockerHost = [](std::string value) {
        std::replace(value.begin(), value.end(), '\\', '/');
        return value;
      };

      auto isWindowsAbsolutePath = [](const std::string& value) {
        return value.size() >= 3 &&
               std::isalpha(static_cast<unsigned char>(value[0])) != 0 &&
               value[1] == ':' &&
               (value[2] == '/' || value[2] == '\\');
      };

      if (configured && *configured) {
        std::string configured_value = configured;
        if (isWindowsAbsolutePath(configured_value))
          return normalizeForDockerHost(configured_value);

        sessions_path = fs::path(configured_value);
        if (sessions_path.is_absolute())
          return normalizeForDockerHost(sessions_path.string());
      } else {
        const fs::path cwd = fs::current_path();
        const fs::path cwd_candidate = cwd / "sessions";
        const fs::path parent_candidate = cwd.parent_path() / "sessions";

        if (fs::exists(cwd_candidate) || cwd.filename() == "asset-broker")
          sessions_path = cwd_candidate;
        else
          sessions_path = parent_candidate;
      }

      std::error_code ec;
      fs::create_directories(sessions_path, ec);

      const auto absolute = fs::absolute(sessions_path, ec);
      if (ec)
        return normalizeForDockerHost(sessions_path.string());
      return normalizeForDockerHost(absolute.string());
    }

    std::string resolvePluginsSourcePath()
    {
      namespace fs = std::filesystem;

      const char* configured = std::getenv("ASSET_BROKER_PLUGINS_DIR");
      fs::path plugins_path;

      auto normalizeForDockerHost = [](std::string value) {
        std::replace(value.begin(), value.end(), '\\', '/');
        return value;
      };

      auto isWindowsAbsolutePath = [](const std::string& value) {
        return value.size() >= 3 &&
               std::isalpha(static_cast<unsigned char>(value[0])) != 0 &&
               value[1] == ':' &&
               (value[2] == '/' || value[2] == '\\');
      };

      if (configured && *configured) {
        std::string configured_value = configured;
        if (isWindowsAbsolutePath(configured_value))
          return normalizeForDockerHost(configured_value);

        plugins_path = fs::path(configured_value);
        if (plugins_path.is_absolute())
          return normalizeForDockerHost(plugins_path.string());
      } else {
        const fs::path cwd = fs::current_path();
        const fs::path cwd_candidate = cwd / "enabled-plugins";
        const fs::path parent_candidate = cwd.parent_path() / "enabled-plugins";

        if (fs::exists(cwd_candidate) || cwd.filename() == "asset-broker")
          plugins_path = cwd_candidate;
        else
          plugins_path = parent_candidate;
      }

      std::error_code ec;
      fs::create_directories(plugins_path / "protocols", ec);

      const auto absolute = fs::absolute(plugins_path, ec);
      if (ec)
        return normalizeForDockerHost(plugins_path.string());
      return normalizeForDockerHost(absolute.string());
    }

    std::string resolvePluginsRuntimePath()
    {
      namespace fs = std::filesystem;

      auto hasProtocolsDir = [](const fs::path& base_path) {
        std::error_code ec;
        return fs::exists(base_path / "protocols", ec) && !ec;
      };

      const char* runtime_dir = std::getenv("ASSET_BROKER_PLUGINS_RUNTIME_DIR");
      if (runtime_dir && *runtime_dir) {
        const fs::path configured(runtime_dir);
        if (hasProtocolsDir(configured))
          return configured.string();
      }

      const fs::path mounted_plugins("/plugins");
      if (hasProtocolsDir(mounted_plugins))
        return mounted_plugins.string();

      const fs::path mounted_plugin("/plugin");
      if (hasProtocolsDir(mounted_plugin))
        return mounted_plugin.string();

      return resolvePluginsSourcePath();
    }

    bool isSharedObjectPlugin(const std::filesystem::path& path)
    {
      auto extension = path.extension().string();
      std::transform(extension.begin(), extension.end(), extension.begin(), [](unsigned char c) {
        return static_cast<char>(std::tolower(c));
      });
      return extension == ".so" || extension == ".dll" || extension == ".dylib";
    }

    json scanProtocolPluginsDirectory(const std::filesystem::path& protocols_path)
    {
      namespace fs = std::filesystem;

      std::error_code ec;
      if (!fs::exists(protocols_path, ec) || ec)
        return json::array();

      json result = json::array();
      for (const auto& entry: fs::directory_iterator(protocols_path, ec)) {
        if (ec)
          break;

        if (!entry.is_regular_file() || !isSharedObjectPlugin(entry.path()))
          continue;

        const std::string protocol = normalizeProtocolName(entry.path().stem().string());
        if (!protocol.empty())
          result.push_back(protocol);
      }

      std::sort(result.begin(), result.end());
      result.erase(std::unique(result.begin(), result.end()), result.end());
      return result;
    }

    class ProtocolPluginsHotReloadWatcher: public HotReloadSystem
    {
      public:
        ProtocolPluginsHotReloadWatcher(
          const std::string& directory,
          std::function<void()> on_change):
          HotReloadSystem(directory),
          _onChange(std::move(on_change))
        {
        }

      protected:
        void onFileCreated(const std::string& filename) override
        {
          if (isSharedObjectPlugin(std::filesystem::path(filename)))
            _onChange();
        }

        void onFileModified(const std::string& filename) override
        {
          if (isSharedObjectPlugin(std::filesystem::path(filename)))
            _onChange();
        }

        void onFileDeleted(const std::string& filename) override
        {
          if (isSharedObjectPlugin(std::filesystem::path(filename)))
            _onChange();
        }

      private:
        std::function<void()> _onChange;
    };

    class ProtocolPluginsRegistry
    {
      public:
        static ProtocolPluginsRegistry& instance()
        {
          static ProtocolPluginsRegistry registry;
          return registry;
        }

        json list()
        {
          ensureStarted();

          std::lock_guard<std::mutex> lock(_mutex);
          return _plugins;
        }

      private:
        ProtocolPluginsRegistry() = default;

        ~ProtocolPluginsRegistry()
        {
          if (_watcher)
            _watcher->stop();
        }

        void ensureStarted()
        {
          std::lock_guard<std::mutex> lock(_mutex);
          if (_started)
            return;

          _protocolsPath = std::filesystem::path(resolvePluginsRuntimePath()) / "protocols";
          _plugins = scanProtocolPluginsDirectory(_protocolsPath);

          std::error_code ec;
          if (std::filesystem::exists(_protocolsPath, ec) && !ec) {
            _watcher = std::make_unique<ProtocolPluginsHotReloadWatcher>(
              _protocolsPath.string(),
              [this]() { this->refresh(); });
            _watcher->start();
          }

          _started = true;
        }

        void refresh()
        {
          std::lock_guard<std::mutex> lock(_mutex);
          _plugins = scanProtocolPluginsDirectory(_protocolsPath);
        }

      private:
        std::mutex _mutex;
        bool _started = false;
        std::filesystem::path _protocolsPath;
        json _plugins = json::array();
        std::unique_ptr<ProtocolPluginsHotReloadWatcher> _watcher;
    };

    // load all running listeners (containers starting with "listener_")
    void ListenerSystem::loadListeners()
    {
        log_info("Loading listeners...");

        ListenerSystem::get()._listenersIds.clear();

        auto output = DockerAPI::DockerHandler::dockerRawGet(
          "http://localhost/containers/json?all=true");

        if (std::get<0>(output) != CURLE_OK)
            throw_exception(DockerAPI::CurlError,
                            std::string("Invalid output from curl: ") +
                              std::to_string(std::get<0>(output)));

        if (std::get<1>(output) != 200)
            throw_exception(DockerAPI::CurlError,
                            std::string("Invalid curl HTTP code: ") +
                              std::to_string(std::get<1>(output)));

        json result = json::array();
        json input  = json::parse(std::get<2>(output));

        for (const auto& el: input) {
            if (el["Names"].get<std::vector<std::string>>()[0].rfind(
                  "/listener_",
                  0) == 0) {
                log_debug("Loaded listener with id: " +
                          el["Id"].get<std::string>());

                ListenerSystem::get()._listenersIds.push_back(
                  el["Id"].get<std::string>());
            }
        }

        log_debug("Loaded containers");
    }

    json ListenerSystem::listListeners()
    {
        auto output = DockerAPI::DockerHandler::dockerRawGet(
          "http://localhost/containers/json?all=true");

        if (std::get<0>(output) != CURLE_OK) {
            throw_exception(
              ListenerError,
              std::string("Failed to list listeners: Curl Error Code: ") +
                std::to_string(std::get<0>(output)));
        }

        if (std::get<1>(output) != 200) {
            throw_exception(
              ListenerError,
              std::string("Failed to list listeners: Http Error Code: ") +
                std::to_string(std::get<1>(output)));
        }

        json input  = json::parse(std::get<2>(output));
        json result = json::array();

        for (const auto& el: input) {
            if (!el.contains("Names") ||
                !el["Names"].is_array() ||
                el["Names"].empty()) {
                continue;
            }

            const auto name = el["Names"][0].get<std::string>();
            if (name.rfind("/listener_", 0) != 0) {
                continue;
            }

            json item;
            item["id"]      = el.value("Id", "");
            item["name"]    = name.size() > 1 ? name.substr(1) : name;
            item["image"]   = el.value("Image", "");
            item["state"]   = el.value("State", "");
            item["status"]  = el.value("Status", "");
            item["created"] = el.value("Created", 0);
            item["ports"]   = el.contains("Ports") ? el["Ports"] : json::array();

            result.push_back(item);
        }

        return result;
    }

    json ListenerSystem::getListener(const std::string& id)
    {
        auto output = DockerAPI::DockerHandler::dockerRawGet(
          "http://localhost/containers/" + id + "/json");

        if (std::get<0>(output) != CURLE_OK) {
            throw_exception(
              ListenerError,
              std::string("Failed to get listener: Curl Error Code: ") +
                std::to_string(std::get<0>(output)));
        }

        if (std::get<1>(output) == 404) {
            throw_exception(ListenerError, "Listener not found");
        }

        if (std::get<1>(output) / 100 != 2) {
            throw_exception(
              ListenerError,
              std::string("Failed to get listener: Http Error Code: ") +
                  std::to_string(std::get<1>(output)) +
                  std::string("; Response: ") +
                  std::get<2>(output));
        }

        json input = json::parse(std::get<2>(output));
        const auto name = input.value("Name", "");
        if (name.rfind("/listener_", 0) != 0) {
            throw_exception(ListenerError, "Listener not found");
        }

        json result;
        result["id"]     = input.value("Id", "");
        result["name"]   = name.size() > 1 ? name.substr(1) : name;
        result["image"]  = input.contains("Config") ? input["Config"].value("Image", "") : "";
        result["state"]  = input.contains("State") ? input["State"].value("Status", "") : "";
        result["status"] = input.contains("State") ? input["State"].value("Status", "") : "";
        result["ports"]  = input.contains("HostConfig") && input["HostConfig"].contains("PortBindings")
                              ? input["HostConfig"]["PortBindings"]
                              : json::object();

        return result;
    }

    /// creates a container with
    std::string ListenerSystem::createListener(
      const std::string&                         plugin_name,
      const std::pair<std::string, std::string>& port_config,
      const std::string&                         protocol_host,
      const std::string&                         session_id)
    {
        log_info("Creating listener...");

        log_debug(std::string("Arguments: plugin_name: \"") +
                  plugin_name +
                  "\"; port_config: {" +
                  port_config.first +
                  ", " +
                  port_config.second +
                  "}");

        std::string listener_name =
          std::string("listener_") + plugin_name + "-" + port_config.second;
        std::replace(listener_name.begin(), listener_name.end(), '/', '-');

        const std::string protocol = normalizeProtocolName(plugin_name);
        const std::string protocol_port = dockerPortToNumeric(port_config.first);
        const std::string resolved_session_id =
          session_id.empty() ? sanitizeSessionId(listener_name) : sanitizeSessionId(session_id);
        const std::string sessions_source_path = resolveSessionsSourcePath();
        const std::string plugins_source_path = resolvePluginsSourcePath();
        const std::string compose_project_name = resolveComposeProjectName();

        std::string protocol_bind_host = "0.0.0.0";
        std::string protocol_target_host = protocol_host;
        const std::string protocol_key = underscoredProtocolName(protocol);

        if (protocol_key == "tcp_bind") {
            if (!protocol_host.empty())
                protocol_bind_host = protocol_host;
        } else if (protocol_target_host.empty()) {
            protocol_target_host = "host.docker.internal";
        }

        json j = {
            {"Image",            "listener:latest"                            },
          {"Labels",
           {{"com.docker.compose.project", compose_project_name},
            {"com.docker.compose.service", "dynamic-listener"},
            {"asset-broker.listener.managed", "true"}}},
            {"Env",
             json::array({"LISTENER_NAME=" + listener_name,
                          "SESSION_ID=" + resolved_session_id,
                          "PROTOCOL=" + protocol,
                          "PROTOCOL_PORT=" + protocol_port,
                          "SESSIONS_ROOT=/sessions",
                          "PROTOCOLS_DIR=/plugin/protocols",
                          "PROTOCOL_BIND=" + protocol_bind_host,
                          "PROTOCOL_HOST=" + protocol_target_host})},

            {"HostConfig",
             {{"PortBindings",
               {{port_config.first,
                 json::array({{{"HostPort", port_config.second}}})}}},
              {"Mounts",
               json::array({{{"Type", "bind"},
                             {"Source", sessions_source_path},
                             {"Target", "/sessions"}},
                            {{"Type", "bind"},
                             {"Source", plugins_source_path},
                             {"Target", "/plugin"}}})},
                {"NetworkMode", "asset-broker-listeners"}}                      },

            {"NetworkingConfig",
             {{"EndpointsConfig",
               {{"asset-broker-listeners",
                 {{"Aliases", json::array({listener_name})}}}}}}              }
        };

        auto output = DockerAPI::DockerHandler::dockerRawPost(
          "http://localhost/containers/create?name=" + listener_name,
          j.dump());

        if (std::get<0>(output) != CURLE_OK) {
            throw_exception(
              ListenerError,
              std::string(
                "Failed to create listener container: Curl Error Code: ") +
                std::to_string(std::get<0>(output)));
        }

        if (std::get<1>(output) / 100 != 2) {
            throw_exception(
              ListenerError,
              std::string(
                "Failed to create listener container: Http Error Code: ") +
                std::to_string(std::get<1>(output)) +
                std::string("; Response: ") +
                std::get<2>(output));
        }

        json data = json::parse(std::get<2>(output));

        auto start_output = DockerAPI::DockerHandler::dockerRawPost(
          "http://localhost/containers/" +
            data["Id"].get<std::string>() +
            "/start",
          "");

        if (std::get<0>(start_output) != CURLE_OK) {
            throw_exception(
              ListenerError,
              std::string(
                "Failed to start listener container: Curl Error Code: ") +
                std::to_string(std::get<0>(start_output)));
        }

        if (std::get<1>(start_output) / 100 != 2) {
            throw_exception(
              ListenerError,
              std::string(
                "Failed to start listener container: Http Error Code: ") +
                std::to_string(std::get<1>(start_output)) +
                std::string("; Response: ") +
                std::get<2>(start_output));
        }

        return data["Id"];
    }

    void ListenerSystem::deleteListener(const std::string& id)
    {
        log_info("Deleting listener...");

        log_debug(std::string("Arguments: id: \"") + id + "\"");

        auto output = DockerAPI::DockerHandler::dockerRawDelete(
          "http://localhost/containers/" + id + "?force=true");

        if (std::get<0>(output) != CURLE_OK) {
            throw_exception(
              ListenerError,
              std::string(
                "Failed to delete listener container: Curl Error Code: ") +
                std::to_string(std::get<0>(output)));
        }

        if (std::get<1>(output) / 100 != 2) {
            throw_exception(
              ListenerError,
              std::string(
                "Failed to delete listener container: Http Error Code: ") +
                std::to_string(std::get<1>(output)));
        }

        log_info(std::string("Listener container deleted successfully: ") + id);
    }

    void ListenerSystem::deleteListeners()
    {
      log_info("Deleting all listener containers...");

      auto output = DockerAPI::DockerHandler::dockerRawGet(
        "http://localhost/containers/json?all=true");

      if (std::get<0>(output) != CURLE_OK) {
        throw_exception(
          ListenerError,
          std::string("Failed to list listeners for cleanup: Curl Error Code: ") +
          std::to_string(std::get<0>(output)));
      }

      if (std::get<1>(output) / 100 != 2) {
        throw_exception(
          ListenerError,
          std::string("Failed to list listeners for cleanup: Http Error Code: ") +
          std::to_string(std::get<1>(output)) +
          std::string("; Response: ") +
          std::get<2>(output));
      }

      json input = json::parse(std::get<2>(output));
      for (const auto &el: input) {
        if (!el.contains("Names") || !el["Names"].is_array() || el["Names"].empty())
          continue;

        const auto name = el["Names"][0].get<std::string>();
        if (name.rfind("/listener_", 0) != 0)
          continue;

        const auto id = el.value("Id", "");
        if (id.empty())
          continue;

        try {
          deleteListener(id);
        } catch (const Flint::Exceptions::Exception &error) {
          log_error(std::string("Failed to delete listener container ") + id + ": " + error.what());
        } catch (const std::exception &error) {
          log_error(std::string("Failed to delete listener container ") + id + ": " + error.what());
        }
      }

      log_info("Listener container cleanup finished");
    }
